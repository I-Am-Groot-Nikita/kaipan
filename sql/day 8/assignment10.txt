Write queries for following requirements:

	Table - I :	officestaff

Q-1	List Name, Designation and Salary of all "officestaff" members.

select name as Name,designation as Designation,salary as Salary from officestaff_trainee_35;

Q-2	List Name and Salary of all the people who are working as "Programmer".

select name "Name", salary "Salary" from officestaff_trainee_35 where designation = "Programmer";

Q-3	List all the information about those people who are earning less than Rs.10,000/- .

select * from officestaff_trainee_35 ot
left join address_trainee_35 at on ot.addressid = at.addressid
where salary < 10000.00 ;

Q-4	List all the information about those staff members who fall in the salary range of 	Rs.5000/- to Rs.10,000/- .

select * from officestaff_trainee_35 ot
left join address_trainee_35 at on ot.addressid = at.addressid
where salary between 50000 and 100000 ;

Q-5	List Name of all the office staff members in Lower Case.

select lower(name) " Name" from officestaff_trainee_35;

Q-6	List Name of all the office staff members in following format:
	For example, if name is "umeshbhai" then display it in the format "Umeshbhai".

select concat(upper(substr(name,1,1)),lower(substr(name,2))) from officestaff_trainee_35;

Q-7	List all the information about those people who works for "Nileshbhai".

select * from officestaff_trainee_35 ot
left join address_trainee_35 at on ot.addressid=at.addressid
where bossno in (select recid from officestaff_trainee_35 where name = "Nileshbhai");

Q-8	List all the information about those members with salary greater than or equal to Rs.3000/- or working directly under "Nileshbhai".

select * from officestaff_trainee_35 ot
left join address_trainee_35 at on ot.addressid=at.addressid
where bossno in (select recid from officestaff_trainee_35 where name = "Nileshbhai") or salary >= 3000;

Q-9	List Name, Salary, Over-time Salary of all the office staff members.

select name "Name", salary "Salary",overtimesalary "Over-time Salary" from officestaff_trainee_35;

Q-10	Perform above MySQL statement again, but now 0(Zero) should be substituted for NULL in 	Over-time Salary field.

select name "Name", salary "Salary",ifnull(overtimesalary,0) "Over-time Salary" from officestaff_trainee_35 ;

Q-11	List Name and Total Monthly Salary (Salary + Over-time Salary) of all the office staff 	members.

select name "Name", (salary + overtimesalary)  "Total Monthly Salary" from officestaff_trainee_35;

Q-12	List Name and Total Annual Salary of all the office staff members.

select name "Name", ((salary + overtimesalary)*12)  "Total Annual Salary" from officestaff_trainee_35;

Q-13	List Name and Total Daily Salary of all the office staff members. (Consider a month is 	of 30 days).

select name "Name", ((salary + overtimesalary)/30)  "Total Daily Salary" from officestaff_trainee_35;

Q-14	List all the information of the person earning Highest Salary (Salary + Over-time 	Salary) in the staff.

select * from officestaff_trainee_35 ot
left join address_trainee_35 at on ot.addressid = at.addressid
where salary+overtimesalary = (select max(salary + overtimesalary) from officestaff_trainee_35) ;

Q-15	List all the information of the person earning Least Salary (Salary + Over-time Salary) 	in the staff.

select * from officestaff_trainee_35 ot
left join address_trainee_35 at on ot.addressid = at.addressid
where salary+overtimesalary = (select min(salary + overtimesalary) from officestaff_trainee_35) ;

Q-16	List all the information about those who earn the Highest Salary (Salary + Over-time 	Salary) and those who earn the Least Salary (Salary + Over-time Salary) in the staff.

select * from officestaff_trainee_35 ot
left join address_trainee_35 at on ot.addressid = at.addressid
where (salary+overtimesalary) =
(select min(salary + overtimesalary) from officestaff_trainee_35)
or salary+overtimesalary = (select max(salary + overtimesalary) from officestaff_trainee_35) ;

Q-17	List all the information about the person who earns more than his boss.

select * from officestaff_trainee_35 ot
left join address_trainee_35 at on ot.addressid = at.addressid
where salary > (select salary from officestaff_trainee_35 where designation = "Boss");

Q-18	List Name, Designation, Salary and City of each staff member.

select ot.name "Name",ot.designation "Designation", ot.salary "Salary", at.city "City" from officestaff_trainee_35 ot
left join address_trainee_35 at on ot.addressid = at.addressid;

Q-19	Using "address" table list total number of persons residing in "Ahmedabad" city.

select count(*) "total number of persons residing in Ahmedabad city" from address_trainee_35 where city = "ahmedabad";

Q-20	List those City Name where 3 or more than 3 people resides.

select city from (select count(*)as counts,city from officestaff_trainee_35 ot
left join address_trainee_35 at on ot.addressid = at.addressid group by city)as t where t.counts > 3;
